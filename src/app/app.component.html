ngrx-carousel

Description

An amazing responsive carousel for Angular projects. Which supports auto play, manual controls with lot of animations

Git repo https://github.com/richardmtp/ngrx-carousel/

How to use?

* npm i ngrx-carousel

* Import NgrxCarouselModule in your app.module.ts 

          import < NgrxCarouselModule > from 'ngrx-carousel';
          imports: [
            ...,
            NgrxCarouselModule
          ],

* add ngrx-carousel tag app.module.html 

          <ngrx-carousel [items]="items" [setting]="setting"></ngrx-carousel>

* app.module.ts import following items

          import < NgrxCarouselItem, NgrxCarouselSetting, NgrxAnimationType > from 'ngrx-carousel';

          items: Array<NgrxCarouselItem> = [];

          setting: NgrxCarouselSetting;

          export class AppComponent {
            constructor() {
              this.items = []
              this.items.push({src: 'assets/image1.png'})
              this.items.push({src: 'assets/image2.png'})
              this.items.push({src: 'assets/image3.png'})

              this.setting = {
                autoPlay: true,
                showArrow: true,
                holdTime: 5000,
                showDots: false,
                animation: NgrxAnimationType.FADE_IN_LEFT
              }
            }
          }

MANUAL CONTROL

Eg:
 
        // #manualControl : This is 'Reference variable' which is used to control ngrx-carousel as a @ViewChild
        <ngrx-carousel #manualControl [items]="items" [setting]="setting"></ngrx-carousel> 
      
        // Trigger button click

        import < NgrxCarouselComponent > from 'ngrx-carousel';
        @ViewChild('manualControl', {static: true}) manualControl: NgrxCarouselComponent;

        playNext() {
          this.manualControl.playNext()
        }

        playPrev() {
          this.manualControl.playPrev()
        }
      
      setting = {
        autoPlay: false,
        showArrow: false,
        holdTime: 5000,
        showDots: true,
        animation: NgrxAnimationType.ZOOM_IN
      }
