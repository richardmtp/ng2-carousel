ngx-carousel

Description

An amazing responsive carousel for Angular projects. Which supports auto play, manual controls with lot of animations

Git repo https://github.com/richardmtp/ngx-carousel/

How to use?

* npm i ngx-carousel

* Import NgxCarouselModule in your app.module.ts 

          import < NgxCarouselModule > from 'ngx-carousel';
          imports: [
            ...,
            NgxCarouselModule
          ],

* add ngx-carousel tag app.module.html 

          <ngx-carousel [items]="items" [setting]="setting"></ngx-carousel>

* app.module.ts import following items

          import < NgxCarouselItem, NgxCarouselSetting, NgxAnimationType > from 'ngx-carousel';

          items: Array<NgxCarouselItem> = [];

          setting: NgxCarouselSetting;

          export class AppComponent {
            constructor() {
              this.items = []
              this.items.push({src: 'assets/image1.png'})
              this.items.push({src: 'assets/image2.png'})
              this.items.push({src: 'assets/image3.png'})

              this.setting = {
                autoPlay: true,
                showArrow: true,
                holdTime: 5000,
                showDots: false,
                animation: NgxAnimationType.FADE_IN_LEFT
              }
            }
          }

MANUAL CONTROL

Eg:
 
        // #manualControl : This is 'Reference variable' which is used to control ngx-carousel as a @ViewChild
        <ngx-carousel #manualControl [items]="items" [setting]="setting"></ngx-carousel> 
      
        // Trigger button click

        import < NgxCarouselComponent > from 'ngx-carousel';
        @ViewChild('manualControl', {static: true}) manualControl: NgxCarouselComponent;

        playNext() {
          this.manualControl.playNext()
        }

        playPrev() {
          this.manualControl.playPrev()
        }
      
      setting = {
        autoPlay: false,
        showArrow: false,
        holdTime: 5000,
        showDots: true,
        animation: NgxAnimationType.ZOOM_IN
      }
